# useScrolling

A hook for components that care about whether or not a user is scrolling.

## Usage

```jsx
import {useScrolling} from '@hzcore/scroll-monitor';

export const ScrollingIndicator = () => {
  const [scrolling, ref] = useScrolling();
  return (
    <div ref={ref} style={scrolling ? styles.scrolling : null}>
      {scrolling ? 'scrolling' : 'not scrolling'}
    </div>
  );
};
```
